// ── YOUTUBE EMBED SYSTEM ──
function renderEmbedPanel(panel){
  if(panel.dataset.rendered)return;
  panel.dataset.rendered='1';
  const key=panel.dataset.ytkey;
  const saved=localStorage.getItem(key);
  if(saved){panel.innerHTML=buildEmbedHTML(saved,key);}
  else{panel.innerHTML=buildInputHTML(key);}
}
function buildInputHTML(key){
  return`<div class="yt-input-area">
    <p class="yt-hint">📺 Paste a YouTube playlist or video URL to embed it. Saved automatically.</p>
    <div class="yt-input-row">
      <input type="text" id="inp_${key}" placeholder="https://www.youtube.com/playlist?list=..."/>
      <button class="primary" onclick="embedYT('${key}')">Embed</button>
    </div>
  </div>`;
}
function buildEmbedHTML(url,key){
  const src=ytToEmbed(url);
  return`<div class="yt-embed-result"><iframe src="${src}" allowfullscreen loading="lazy"></iframe></div>
  <div class="yt-change-row">
    <p>📎 <a href="${url}" target="_blank" style="font-size:11px;word-break:break-all;">${url}</a></p>
    <button class="small" onclick="clearEmbed('${key}')">× Change</button>
  </div>`;
}
function ytToEmbed(url){
  try{
    const u=new URL(url);
    const list=u.searchParams.get('list');
    const v=u.searchParams.get('v');
    if(list)return`https://www.youtube.com/embed/videoseries?list=${list}`;
    if(v)return`https://www.youtube.com/embed/${v}`;
    const path=u.pathname.split('/');
    const id=path[path.length-1];
    if(id)return`https://www.youtube.com/embed/${id}`;
  }catch(e){}
  return url;
}
function embedYT(key){
  const inp=document.getElementById('inp_'+key);
  if(!inp||!inp.value.trim())return;
  const url=inp.value.trim();
  localStorage.setItem(key,url);
  const panel=document.querySelector(`[data-ytkey="${key}"]`);
  if(panel){panel.dataset.rendered='1';panel.innerHTML=buildEmbedHTML(url,key);}
}
function clearEmbed(key){
  localStorage.removeItem(key);
  const panel=document.querySelector(`[data-ytkey="${key}"]`);
  if(panel){delete panel.dataset.rendered;panel.innerHTML=buildInputHTML(key);}
}

// ── DEADLINE TRACKER ──
let deadlines=JSON.parse(localStorage.getItem('deadlines')||'[]');
function daysUntil(d){
  if(!d)return 999;
  return Math.ceil((new Date(d)-new Date())/(1000*60*60*24));
}
function urgencyClass(d){
  const n=daysUntil(d);
  if(n<0)return{cls:'urgent',txt:'⚠️ Overdue'};
  if(n<=7)return{cls:'urgent',txt:`🔴 ${n} day${n===1?'':'s'} left`};
  if(n<=21)return{cls:'soon',txt:`🟡 ${n} days left`};
  return{cls:'safe',txt:`🟢 ${n} days left`};
}
function renderDeadlines(containerId){
  const c=document.getElementById(containerId);
  if(!c)return;
  c.innerHTML='';
  if(!deadlines.length){
    c.innerHTML='<p style="font-size:12px;color:var(--muted);padding:6px 0;">No deadlines added yet.</p>';
    return;
  }
  deadlines.forEach((d,i)=>{
    const u=urgencyClass(d.date);
    const card=document.createElement('div');
    card.className='deadline-card';
    card.innerHTML=`
      <div class="deadline-title">${d.title}</div>
      <div class="deadline-meta">📅 ${d.date||'No date'}${d.note?' · '+d.note:''}</div>
      <div class="deadline-status ${u.cls}">${u.txt}</div>
      <button class="small" style="margin-top:7px;" onclick="removeDeadline(${i})">✕ Remove</button>`;
    c.appendChild(card);
  });
}
function addDeadline(){
  const title=document.getElementById('dl-title').value.trim();
  const date=document.getElementById('dl-date').value;
  const note=document.getElementById('dl-note').value.trim();
  if(!title)return;
  deadlines.push({title,date,note});
  localStorage.setItem('deadlines',JSON.stringify(deadlines));
  renderDeadlines('deadline-container');
  renderDeadlines('cal-deadline-container');
  document.getElementById('dl-title').value='';
  document.getElementById('dl-date').value='';
  document.getElementById('dl-note').value='';
}
